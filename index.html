3. Contoh Kode Integrasi (JavaScript Fetch)

fetch('https://albitiktok.vercel.app/?api=true&url=https://vt.tiktok.com/ZS9PW3Bvm/')
  .then(res => res.json())
  .then(data => {
    console.log("Judul Video:", data.data.title);
    console.log("Video No Watermark:", data.data.play);
    console.log("Audio MP3:", data.data.music);
  })
  .catch(err => console.error(err));
